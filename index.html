            @keyframes color {
                0% {
                    transform: rotateX(0deg);

                }
            }

            @keyframes rotate {
                0% {
                    transform: rotateY(0deg);
                }
                100% {
                    transform: rotateY(360deg);
                }
            }

              color: #f00;
              animation: color 32s linear infinite;
            }

              color: #333;
              transform: rotateY(30deg);
